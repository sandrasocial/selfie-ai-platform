import { useState, useEffect } from 'react';
import { Link, useLocation } from 'wouter';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import Header from '@/components/layout/Header';
import Footer from '@/components/layout/Footer';
import { useMutation } from '@tanstack/react-query';
import { useToast } from '@/hooks/use-toast';

export default function OfferDownsell() {
  const [, setLocation] = useLocation();
  const [user, setUser] = useState<any>(null);
  const { toast } = useToast();

  useEffect(() => {
    const loadUserData = async () => {
      try {
        const userResponse = await fetch("/api/me", { credentials: "include" });
        if (userResponse.ok) {
          const userData = await userResponse.json();
          setUser(userData);
        }
      } catch (error) {
        console.log("User not authenticated");
      }
    };
    loadUserData();
  }, []);

  const downsellMutation = useMutation({
    mutationFn: async (action: 'accept' | 'decline') => {
      const response = await fetch("/api/downsell-response", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ action }),
      });
      if (!response.ok) throw new Error("Failed to process downsell");
      return response.json();
    },
    onSuccess: (data) => {
      if (data.action === 'accept') {
        window.location.href = data.paymentUrl || "/thank-you";
      } else {
        setLocation("/dashboard");
      }
    },
    onError: (error: any) => {
      toast({
        title: "Error",
        description: error.message || "Something went wrong",
        variant: "destructive",
      });
    },
  });

  const handleDownsell = (action: 'accept' | 'decline') => {
    downsellMutation.mutate(action);
  };

  return (
    <div className="min-h-screen bg-white">
      <Header user={user} />
      <main className="container mx-auto px-6 py-24">
        <div className="text-center mb-20">
          <p className="font-['Neue Einstellung'] uppercase text-sm tracking-[0.15em] text-gray-700 mb-4">
            Last Chance Offer
          </p>
          <h1 className="font-['Cormorant Garamond'] text-5xl text-black mb-6">
            Wait — Don't Leave Empty-Handed
          </h1>
          <p className="text-lg font-['Neue Einstellung'] text-gray-700 max-w-2xl mx-auto">
            Not ready for the full program? Take the first step with this premium starter kit and start showing up now.
          </p>
        </div>

        <div className="max-w-4xl mx-auto">
          <Card className="bg-[#FAFAF9] border border-gray-200">
            <CardHeader className="text-center">
              <CardTitle className="font-['Cormorant Garamond'] text-4xl text-black mb-4">
                Selfie Starter Kit
              </CardTitle>
              <div className="text-black text-3xl font-['Neue Einstellung'] mb-2">$27</div>
              <div className="text-gray-500 text-sm line-through font-['Neue Einstellung'] mb-1">$97</div>
              <div className="text-xs font-['Neue Einstellung'] uppercase tracking-widest text-red-600 mb-4">
                72% Off Today Only
              </div>
              <p className="text-sm text-gray-600 font-['Neue Einstellung']">One-time payment • Instant Access</p>
            </CardHeader>

            <CardContent className="pt-8 pb-12 px-8">
              <h3 className="text-lg text-black font-semibold text-center mb-6 font-['Neue Einstellung']">
                Inside Your Kit:
              </h3>
              <ul className="grid md:grid-cols-2 gap-4 text-sm text-gray-800 font-['Neue Einstellung'] mb-10">
                <li>— 10 Viral Selfie Poses Guide</li>
                <li>— Perfect Lighting Setup Tutorial</li>
                <li>— Angle & Composition Cheat Sheet</li>
                <li>— Phone Camera Settings Masterclass</li>
                <li>— 5 Editing Presets for Instagram</li>
                <li>— Caption Templates That Convert</li>
              </ul>

              <div className="text-center bg-white border border-gray-200 p-4 mb-8">
                <p className="text-xs font-['Neue Einstellung'] text-gray-700 tracking-wide uppercase">
                  This offer disappears when you exit this page
                </p>
              </div>

              <div className="space-y-4">
                <Button
                  onClick={() => handleDownsell('accept')}
                  disabled={downsellMutation.isPending}
                  className="w-full bg-black text-white rounded-none py-4 font-['Neue Einstellung'] uppercase tracking-wider"
                >
                  {downsellMutation.isPending ? "Processing..." : "Yes, I Want the Starter Kit"}
                </Button>
                <Button
                  onClick={() => handleDownsell('decline')}
                  disabled={downsellMutation.isPending}
                  variant="ghost"
                  className="w-full text-gray-600 font-['Neue Einstellung'] uppercase tracking-wider"
                >
                  No Thanks, Take Me to My Dashboard
                </Button>
              </div>

              <div className="mt-10 text-center">
                <p className="italic text-sm text-gray-600 font-['Neue Einstellung'] max-w-md mx-auto">
                  "The starter kit gave me the confidence to finally post my first selfie. I got over 200 likes." — Maria K.
                </p>
              </div>
            </CardContent>
          </Card>
        </div>
      </main>
      <Footer />
    </div>
  );
}
