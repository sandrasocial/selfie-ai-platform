Complete Implementation Guide
Here's how to implement your enhanced dashboard with AI Brand Hub onboarding:
Step 1: Database Updates
First, update your database schema:
sql-- Add to users table
ALTER TABLE users ADD COLUMN brand_onboarding_complete BOOLEAN DEFAULT FALSE;
ALTER TABLE users ADD COLUMN onboarding_completed_at TIMESTAMP;
ALTER TABLE users ADD COLUMN vip_status BOOLEAN DEFAULT FALSE;

-- Create or update purchases table
CREATE TABLE IF NOT EXISTS purchases (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  product_id VARCHAR(255) NOT NULL,
  stripe_payment_intent_id VARCHAR(255),
  amount INTEGER,
  currency VARCHAR(3) DEFAULT 'usd',
  status VARCHAR(50) DEFAULT 'completed',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create subscriptions table for ongoing memberships
CREATE TABLE IF NOT EXISTS subscriptions (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  stripe_subscription_id VARCHAR(255) UNIQUE,
  tier VARCHAR(50) NOT NULL, -- 'starter', 'branded', 'vip'
  status VARCHAR(50) NOT NULL, -- 'active', 'canceled', 'past_due'
  current_period_start TIMESTAMP,
  current_period_end TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Step 5: Access Control Implementation
Here's how to implement the access control in your AI tools:
javascript// In Sandra AI component
import { useUserAccess } from '@/hooks/useUserAccess';

function SandraAI() {
  const { user, accessLevels, hasProfile } = useUserAccess();
  
  // Block access if no starter access
  if (!accessLevels.canAccessAI) {
    return <UpgradePrompt requiredTier="starter" />;
  }
  
  // Prompt for profile completion if missing
  if (!hasProfile) {
    return <ProfileCompletionPrompt />;
  }
  
  // Regular AI interface
  return <AIInterface />;
}
