You need to:

Wrap the fallback content in res.status(200).json({ ... })

Ensure no HTML/undefined errors sneak in

Log and return the fallback properly

try {
  const aiResponse = await callAnthropicAPI(inputs);
  
  if (!aiResponse || !aiResponse.content) {
    throw new Error("AI response empty");
  }

  return res.status(200).json({
    title: "Your Story-Based Positioning Blueprint",
    content: aiResponse.content, // AI-generated content
  });
} catch (err) {
  console.error("AI failed, using fallback. Error:", err.message);

  return res.status(200).json({
    title: "Your Story-Based Positioning Blueprint",
    fallback: true,
    content: [
      {
        heading: "Step 1: Own Your Rock Bottom",
        text: "Your past isn’t something to hide—it’s the key to building an authentic brand. Share that moment you hit the wall, and how it cracked you open."
