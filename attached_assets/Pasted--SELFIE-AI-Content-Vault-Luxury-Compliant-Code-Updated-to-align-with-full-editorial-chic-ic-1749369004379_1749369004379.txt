import { useState, useEffect } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import Header from '@/components/layout/Header';

interface VaultEntry {
  id: number;
  templateName: string;
  contentText: string;
  hookText?: string;
  hashtags?: string;
  createdAt: string;
}

export default function ContentVault() {
  const [vaultEntries, setVaultEntries] = useState<VaultEntry[]>([]);
  const [selectedDay, setSelectedDay] = useState<number>(1);

  useEffect(() => {
    loadVaultEntries();
  }, []);

  const loadVaultEntries = async () => {
    try {
      const response = await fetch('/api/vault', { credentials: 'include' });
      if (response.ok) {
        const data = await response.json();
        setVaultEntries(data);
      }
    } catch (error) {
      console.error('Failed to load vault entries:', error);
    }
  };

  return (
    <div className="min-h-screen bg-[#F1F1F1] text-[#171719] font-inter">
      <Header />

      {/* Hero Section */}
      <section className="w-full h-[60vh] bg-black text-white flex items-center justify-center text-center px-6">
        <div>
          <h1 className="text-[64px] font-['Cormorant Garamond'] font-normal leading-tight mb-4">The Vault</h1>
          <p className="italic text-[18px] text-[#F1F1F1] max-w-2xl mx-auto">
            Your best content ideas, curated and saved in style.
          </p>
        </div>
      </section>

      {/* Main Vault Section */}
      <section className="py-20 px-6 max-w-6xl mx-auto">
        <div className="mb-12 text-center">
          <h2 className="text-[36px] font-['Cormorant Garamond'] italic mb-4">Saved Content Entries</h2>
          <p className="text-[16px] text-[#4C4B4B]">Review, refine, and export your favorites.</p>
        </div>

        {vaultEntries.length === 0 ? (
          <div className="text-center py-24">
            <p className="italic text-[#4C4B4B]">Your vault is currently empty. Start generating content to see it here.</p>
          </div>
        ) : (
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-10">
            {vaultEntries.map((entry) => (
              <Card key={entry.id} className="bg-white border border-[#B5B5B3] p-8">
                <CardHeader>
                  <CardTitle className="text-[24px] font-['Cormorant Garamond'] mb-2">
                    {entry.templateName}
                  </CardTitle>
                  <p className="text-sm text-[#4C4B4B] italic">
                    {new Date(entry.createdAt).toLocaleDateString()}
                  </p>
                </CardHeader>
                <CardContent className="mt-6 space-y-4">
                  {entry.hookText && (
                    <div>
                      <p className="text-xs uppercase tracking-wide text-[#4C4B4B] mb-1">Hook</p>
                      <p className="text-sm text-[#171719]">{entry.hookText}</p>
                    </div>
                  )}
                  <div>
                    <p className="text-xs uppercase tracking-wide text-[#4C4B4B] mb-1">Content</p>
                    <p className="text-sm text-[#171719] leading-relaxed">{entry.contentText}</p>
                  </div>
                  {entry.hashtags && (
                    <div>
                      <p className="text-xs uppercase tracking-wide text-[#4C4B4B] mb-1">Hashtags</p>
                      <p className="text-xs text-[#4C4B4B]">{entry.hashtags}</p>
                    </div>
                  )}
                </CardContent>
              </Card>
            ))}
          </div>
        )}
      </section>
    </div>
  );
}
