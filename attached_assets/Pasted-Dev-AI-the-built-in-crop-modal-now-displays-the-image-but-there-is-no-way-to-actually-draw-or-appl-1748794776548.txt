       const { x: x1, y: y1 } = cropStart;
       const { x: x2, y: y2 } = cropEnd;
       const cropX = Math.min(x1, x2);
       const cropY = Math.min(y1, y2);
       const cropWidth = Math.abs(x2 - x1);
       const cropHeight = Math.abs(y2 - y1);
       if (cropWidth === 0 || cropHeight === 0) {
         // No area selected; do nothing or show a warning.
         return;
       }
       // Create temporary canvas to extract crop:
       const tempCanvas = document.createElement("canvas");
       const tempCtx = tempCanvas.getContext("2d");
       tempCanvas.width = cropWidth;
       tempCanvas.height = cropHeight;
       // Draw only the selected portion from the main canvas:
       tempCtx.drawImage(
         document.getElementById("cropCanvas"),
         cropX, cropY, cropWidth, cropHeight,
