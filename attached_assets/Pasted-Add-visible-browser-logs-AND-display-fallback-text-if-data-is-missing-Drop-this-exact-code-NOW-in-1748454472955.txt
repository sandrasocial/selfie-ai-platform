useEffect(() => {
  const fetchAnswers = async () => {
    try {
      const res = await fetch('/api/module2-answers');
      if (!res.ok) {
        console.error("❌ Response not OK:", res.status);
        return;
      }
      const data = await res.json();
      console.log("✅ Fetched saved answers:", data);

      if (!data || Object.keys(data).length === 0) {
        alert("⚠️ No saved answers found.");
      }

      setFormData(data);
    } catch (err) {
      console.error("❌ Error fetching saved answers:", err);
    }
  };
  fetchAnswers();
}, []);

const handleGenerate = async () => {
  try {
    const res = await fetch('/api/module2-workbook', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(formData),
    });

    if (!res.ok) {
      console.error("❌ Workbook request failed:", res.status);
      return;
    }

    const data = await res.json();
    console.log("✅ Generated Preview:", data);

    if (!data || !data.pov) {
      alert("⚠️ Preview data is empty. Check server response.");
    }

    setPreviewData(data);
  } catch (err) {
    console.error("❌ Workbook generation error:", err);
  }
};
