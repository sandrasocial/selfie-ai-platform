import React, { useState } from 'react';
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Badge } from "@/components/ui/badge";
import { Link } from 'wouter';
import CustomUploadButton from '@/components/CustomUploadButton';
import { PhotoEnhancementActions } from '@/components/PhotoEnhancementActions';
import { usePhotoLibrary } from '@/hooks/usePhotoLibrary';
import { useHybridAuth } from '@/hooks/useHybridAuth';
import { LogoutButton } from '@/components/LogoutButton';
import { useToast } from '@/hooks/use-toast';

export default function PhotoUpload() {
  const [uploadingPhoto, setUploadingPhoto] = useState(false);
  const [selectedTags, setSelectedTags] = useState<string[]>([]);
  const [customTag, setCustomTag] = useState('');
  
  const { user, isAuthenticated, isLoading } = useHybridAuth();
  const photoLibrary = usePhotoLibrary(user?.id || '');
  const { toast } = useToast();

  if (!isLoading && !isAuthenticated) {
    window.location.href = '/supabase-auth';
    return null;
  }

  if (isLoading) {
    return (
      <div className="min-h-screen bg-white flex items-center justify-center">
        <p className="font-['Neue Einstellung'] text-gray-600">Loading...</p>
      </div>
    );
  }

  const predefinedTags = [
    'selfie', 'professional', 'casual', 'outdoor', 'indoor', 
    'headshot', 'lifestyle', 'business', 'portrait', 'candid'
  ];

  const handleUpload = async (fileInfo: any) => {
    setUploadingPhoto(true);
    
    try {
      await photoLibrary.savePhoto({
        uploadcare_uuid: fileInfo.uuid,
        file_url: fileInfo.cdnUrl,
        file_name: fileInfo.name,
        file_size: fileInfo.size,
        tags: selectedTags
      });

      setSelectedTags([]);
      setCustomTag('');
      
      toast({
        title: "Photo Uploaded",
        description: "Your photo has been saved to the library",
      });
    } catch (error) {
      console.error('Upload error:', error);
    } finally {
      setUploadingPhoto(false);
    }
  };

  const toggleTag = (tag: string) => {
    setSelectedTags(prev => 
      prev.includes(tag) 
        ? prev.filter(t => t !== tag)
        : [...prev, tag]
    );
  };

  const addCustomTag = () => {
    if (customTag.trim() && !selectedTags.includes(customTag.trim())) {
      setSelectedTags(prev => [...prev, customTag.trim()]);
      setCustomTag('');
    }
  };

  const removeTag = (tag: string) => {
    setSelectedTags(prev => prev.filter(t => t !== tag));
  };

  return (
    <div className="min-h-screen bg-white">
      <div className="border-b border-gray-100 bg-white">
        <div className="max-w-4xl mx-auto px-6 py-4 flex justify-between items-center">
          <div>
            <h1 className="font-['Cormorant Garamond'] text-3xl text-black mb-1">Upload Photos</h1>
            <p className="font-['Neue Einstellung'] text-gray-500 text-sm">Add new visuals to your SELFIE AI™ library</p>
          </div>
          <div className="text-right">
            <p className="font-['Neue Einstellung'] text-sm text-gray-500">
              {user?.firstName || user?.name || user?.email}
            </p>
            <LogoutButton />
          </div>
        </div>
      </div>

      <div className="max-w-5xl mx-auto px-6 py-12 grid grid-cols-1 lg:grid-cols-2 gap-12">
        <Card className="border border-black">
          <CardHeader>
            <CardTitle className="font-['Cormorant Garamond'] text-2xl text-black">
              Upload New Photo
            </CardTitle>
            <p className="font-['Neue Einstellung'] text-sm text-gray-600">
              Add high-quality photos to elevate your brand identity
            </p>
          </CardHeader>
          <CardContent className="space-y-6">
            <div>
              <Label className="font-['Neue Einstellung'] text-sm mb-2 block">
                Tags (Optional)
              </Label>
              <div className="flex flex-wrap gap-2 mb-4">
                {predefinedTags.map(tag => (
                  <Badge
                    key={tag}
                    variant={selectedTags.includes(tag) ? "default" : "outline"}
                    className={`cursor-pointer uppercase text-xs tracking-wider font-['Neue Einstellung'] ${
                      selectedTags.includes(tag) 
                        ? 'bg-black text-white' 
                        : 'border-gray-300 text-gray-600'
                    }`}
                    onClick={() => toggleTag(tag)}
                  >
                    {tag}
                  </Badge>
                ))}
              </div>

              <div className="flex gap-2">
                <Input
                  placeholder="Custom tag"
                  value={customTag}
                  onChange={(e) => setCustomTag(e.target.value)}
                  onKeyDown={(e) => e.key === 'Enter' && addCustomTag()}
                  className="font-['Neue Einstellung']"
                />
                <Button
                  onClick={addCustomTag}
                  disabled={!customTag.trim()}
                  variant="outline"
                  className="uppercase text-xs font-['Neue Einstellung'] px-4"
                >
                  Add
                </Button>
              </div>

              {selectedTags.length > 0 && (
                <div className="mt-4">
                  <p className="text-xs text-gray-500 mb-2 uppercase font-['Neue Einstellung']">Selected Tags</p>
                  <div className="flex flex-wrap gap-2">
                    {selectedTags.map(tag => (
                      <Badge key={tag} className="bg-black text-white text-xs font-['Neue Einstellung']">
                        {tag}
                        <button
                          onClick={() => removeTag(tag)}
                          className="ml-1"
                        >
                          ×
                        </button>
                      </Badge>
                    ))}
                  </div>
                </div>
              )}
            </div>

            <div className="text-center py-10 border-2 border-dashed border-gray-300 rounded-lg">
              <CustomUploadButton
                onUpload={handleUpload}
                disabled={uploadingPhoto}
              />
              {uploadingPhoto && (
                <p className="font-['Neue Einstellung'] text-sm text-gray-600 mt-3">
                  Uploading and saving...
                </p>
              )}
            </div>
          </CardContent>
        </Card>

        <Card className="border border-gray-200">
          <CardHeader>
            <CardTitle className="font-['Cormorant Garamond'] text-2xl text-black">
              Your Photo Library
            </CardTitle>
            <p className="font-['Neue Einstellung'] text-sm text-gray-600">
              {photoLibrary.photos.length} photos uploaded
            </p>
          </CardHeader>
          <CardContent>
            {photoLibrary.loading ? (
              <div className="text-center py-12">
                <p className="font-['Neue Einstellung'] text-gray-500">Loading...</p>
              </div>
            ) : photoLibrary.photos.length === 0 ? (
              <div className="text-center py-12">
                <p className="font-['Neue Einstellung'] text-gray-600">No photos yet.</p>
                <p className="font-['Neue Einstellung'] text-sm text-gray-400">Upload your first to get started</p>
              </div>
            ) : (
              <div className="grid grid-cols-1 gap-6 max-h-96 overflow-y-auto">
                {photoLibrary.photos.slice(0, 8).map((photo, index) => (
                  <div key={index} className="border border-gray-100 rounded p-4">
                    <div className="flex gap-4">
                      <div className="w-24 h-24 overflow-hidden rounded border border-gray-200">
                        <img
                          src={photo.file_url}
                          alt={photo.file_name}
                          className="w-full h-full object-cover"
                        />
                      </div>
                      <div className="flex-1">
                        <p className="text-sm font-['Neue Einstellung'] text-black truncate">
                          {photo.file_name}
                        </p>
                        <PhotoEnhancementActions
                          imageUrl={photo.file_url}
                          photoId={photo.id?.toString()}
                          compact={true}
                          onEnhanced={(enhancedUrl, type) => {
                            toast({
                              title: "Photo Enhanced",
                              description: `${type.replace('_', ' ')} completed`,
                            });
                          }}
                        />
                      </div>
                    </div>
                    {photo.tags && photo.tags.length > 0 && (
                      <div className="mt-2 flex flex-wrap gap-2">
                        {photo.tags.slice(0, 2).map((tag, idx) => (
                          <Badge key={idx} className="bg-black text-white text-xs font-['Neue Einstellung']">
                            {tag}
                          </Badge>
                        ))}
                        {photo.tags.length > 2 && (
                          <Badge className="bg-gray-600 text-white text-xs font-['Neue Einstellung']">
                            +{photo.tags.length - 2}
                          </Badge>
                        )}
                      </div>
                    )}
                  </div>
                ))}
              </div>
            )}
            {photoLibrary.photos.length > 12 && (
              <div className="text-center mt-6">
                <Link href="/supabase-test">
                  <Button variant="outline" className="uppercase text-xs font-['Neue Einstellung'] px-6">
                    View All
                  </Button>
                </Link>
              </div>
            )}
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
