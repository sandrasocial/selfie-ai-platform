// Access Control Helper
const useUserAccess = (user) => {
  const [accessLevels, setAccessLevels] = useState({
    hasFreeTrial: false,
    trialDaysLeft: 0,
    hasStarterAccess: false,
    hasBrandedAccess: false,
    hasVipAccess: false,
    canAccessAI: false,
    canAccessWorkbooks: false,
    canAccessAdvancedFeatures: false
  });

  useEffect(() => {
    if (user) {
      const now = new Date();
      const purchases = user.purchases || [];
      
      // Check trial periods
      let hasFreeTrial = false;
      let trialDaysLeft = 0;
      
      if (purchases.includes('starter-kit') && user.starterKitPurchaseDate) {
        const purchaseDate = new Date(user.starterKitPurchaseDate);
        const trialEndDate = new Date(purchaseDate.getTime() + (30 * 24 * 60 * 60 * 1000)); // 30 days
        trialDaysLeft = Math.max(0, Math.ceil((trialEndDate - now) / (24 * 60 * 60 * 1000)));
        hasFreeTrial = trialDaysLeft > 0;
      }
      
      if (purchases.includes('branded-by-selfie') && user.brandedPurchaseDate) {
        const purchaseDate = new Date(user.brandedPurchaseDate);
        const trialEndDate = new Date(purchaseDate.getTime() + (90 * 24 * 60 * 60 * 1000)); // 90 days
        trialDaysLeft = Math.max(trialDaysLeft, Math.ceil((trialEndDate - now) / (24 * 60 * 60 * 1000)));
        hasFreeTrial = trialDaysLeft > 0;
      }

      setAccessLevels({
        hasFreeTrial,
        trialDaysLeft,
        hasStarterAccess: purchases.includes('starter-kit') || hasFreeTrial,
        hasBrandedAccess: purchases.includes('branded-by-selfie') || hasFreeTrial,
        hasVipAccess: user.vipStatus || false,
        canAccessAI: purchases.includes('starter-kit') || purchases.includes('branded-by-selfie') || hasFreeTrial || user.vipStatus,
        canAccessWorkbooks: purchases.includes('branded-by-selfie') || hasFreeTrial || user.vipStatus,
        canAccessAdvancedFeatures: user.vipStatus || hasFreeTrial
      });
    }
  }, [user]);

  return accessLevels;
};
