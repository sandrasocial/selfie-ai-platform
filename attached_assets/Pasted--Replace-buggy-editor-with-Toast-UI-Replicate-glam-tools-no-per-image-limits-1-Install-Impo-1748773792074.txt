async function runReplicate(model, payload) {
  const res = await fetch('https://api.replicate.com/v1/predictions', {
    method: 'POST',
    headers: {
      'Authorization': `Token ${process.env.REACT_APP_REPLICATE_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({version: model, input: payload})
  });
  const {output} = await res.json();
  editorRef.current.getInstance().loadImageFromURL(output, 'result');
}
