AI-Integration Plan
Workbook Generator
Flow: User completes module → clicks "Generate Workbook" → AI processes notes + progress → PDF generated → download link provided
jsxconst generateWorkbookMutation = useMutation({
  mutationFn: async ({ templateId, moduleId, workbookType }) => {
    const response = await fetch('/api/ai/workbook', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        templateId,
        moduleId,
        workbookType,
        userNotes: notes[moduleId] || '',
        selectedAesthetic,
        progressData: completedLessons,
        courseId: 'starter-kit'
      })
    });
    return response.json();
  },
  onSuccess: (data) => {
    toast({ title: "Workbook Ready!", description: "Download starting..." });
    window.open(data.downloadUrl, '_blank');
  }
});
