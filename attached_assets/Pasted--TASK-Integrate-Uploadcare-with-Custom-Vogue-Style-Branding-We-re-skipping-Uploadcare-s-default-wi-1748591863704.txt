import uploadcare from 'uploadcare-widget';
import 'uploadcare-widget/uploadcare.lang.en.min.js';

const CustomUploadButton = ({ onUpload }: { onUpload: (url: string) => void }) => {
  const openUploader = () => {
    uploadcare.openDialog(null, {
      publicKey: 'YOUR_PUBLIC_UPLOADCARE_KEY', // Replace with real key
    }).done(file => {
      file.done(fileInfo => {
        if (fileInfo?.cdnUrl) {
          onUpload(fileInfo.cdnUrl);
        }
      });
    });
  };

  return (
    <button
      onClick={openUploader}
      className="rounded-2xl bg-black text-white px-6 py-3 text-sm tracking-wide uppercase shadow-md hover:bg-white hover:text-black border border-black transition duration-300"
    >
      Upload Your Photo
    </button>
  );
};

export default CustomUploadButton;
📌 Replace 'YOUR_PUBLIC_UPLOADCARE_KEY' with the real key Sandra provides.

import CustomUploadButton from '@/components/CustomUploadButton';
import { usePhotoLibrary } from '@/hooks/usePhotoLibrary';
import { useAuth } from '@/hooks/useAuth'; // or wherever user ID comes from

const PhotoUploadPage = () => {
  const { addPhoto } = usePhotoLibrary();
  const { user } = useAuth(); // Make sure user ID is available

  const handleUpload = async (url: string) => {
    await addPhoto({
      image_url: url,
      user_id: user.id,
      category: 'selfie',
      caption: '',
    });
  };

  return (
    <div className="p-6 max-w-xl mx-auto text-center">
      <h1 className="text-2xl font-semibold mb-4">Upload a New Photo</h1>
      <CustomUploadButton onUpload={handleUpload} />
    </div>
  );
};

export default PhotoUploadPage;
